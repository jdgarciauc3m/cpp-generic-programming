\subsection{Overview of standard concepts}

\begin{frame}[t,fragile]{Concepts in the standard library}
\begin{itemize}
  \item Standard \cppkey{concepts} are distributed in several header files.
    \begin{itemize}
      \item \cppid{<concepts>}: Basic concepts.
      \item \cppid{<compare>}: Comparison concepts.
      \item \cppid{<iterator>}: Iterator concepts.
      \item \cppid{<ranges>}: Ranges concepts.
      \item \cppid{<random>}: Random concepts.
    \end{itemize}
\end{itemize}
\end{frame}
